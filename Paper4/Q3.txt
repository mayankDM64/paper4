
#include <iostream>
using namespace std;

int main() {
    int n;
     cout << "enter the size of array " << endl;
     cin >> n;
   
    int arr[n];
    
    cout << "enter the input in the array " << endl;
    
    for (int i = 0; i < n; i++) 
    {cin >> arr[i];
    }
    
   // int arr[]= {2, 5, 3, 5, 4, 4, 2, 3}; {3, 5, 4, 2, 6, 5, 6, 6, 5, 4, 8, 3}
    int x;
    int y;
    cout << "x = " << endl;
     cin >> x;
     cout << "y = " << endl;
     cin >> y;
    int h=0;
    int j=0;
    int dm=n;
    int d;
    
    
    for(int i=0; i<n; i++)
    {
        if(arr[i] == x)
        {
           h=i;
            if(j!=0)
            { d=h-j;
             if(d<0)
                 d=-d;
            }
        }
        if(arr[i] == y)
        {
           j=i;
            if(h!=0)
            { d=h-j;
             if(d<0)
                 d=-d;
            }
        }
        
        if(d<dm)
            dm=d;
    }
    
   cout << "Minimum distance between " << x << " and " << y << " is " << dm;
printf("Integer value is %d\n" , dm);
    
}
